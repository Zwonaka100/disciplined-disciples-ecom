rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Allow all users to read/write their own data
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Allow authenticated users to access their profiles
    match /artifacts/default-app-id/users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Allow authenticated users to manage their orders
    match /orders/{orderId} {
      allow read, write: if request.auth != null && request.auth.uid == resource.data.userId;
    }
    
    // Allow authenticated users to manage their cart
    match /cart/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Allow authenticated users to read products (public data)
    match /products/{productId} {
      allow read: if true; // Public read access for all products
      allow write: if false; // Only admin can write (handle via admin panel)
    }
    
    // Allow test collections for debugging
    match /test/{document} {
      allow read, write: if request.auth != null;
    }
    
    // Deny all other access
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
