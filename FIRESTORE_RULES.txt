// FIREBASE FIRESTORE SECURITY RULES
// Copy these into Firebase Console > Firestore Database > Rules and click Publish.

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function isSignedIn() {
      return request.auth != null;
    }

    function isAdmin() {
      return isSignedIn() && (
        request.auth.token.email == 'zmabege@gmail.com' ||
        request.auth.token.email == 'nomaqhizazolile@gmail.com'
      );
    }

    function isUserSelf(userId) {
      return isSignedIn() && request.auth.uid == userId;
    }

    // Profiles: user can read/write self; admins manage all
    match /artifacts/default-app-id/users/{userId} {
      allow read: if isUserSelf(userId) || isAdmin();
      allow write: if isUserSelf(userId) || isAdmin();
    }

    // Orders: user can create/read/update own; admins manage all; only admins delete
    match /artifacts/default-app-id/orders/{orderId} {
      allow create: if isSignedIn() && request.resource.data.userId == request.auth.uid;
      allow read: if (isSignedIn() && request.auth.uid == resource.data.userId) || isAdmin();
      allow update: if (isSignedIn() && request.auth.uid == resource.data.userId) || isAdmin();
      allow delete: if isAdmin();
    }

    // Support requests: user can manage own; admins manage all
    match /artifacts/default-app-id/support-requests/{requestId} {
      allow create: if isSignedIn() && request.resource.data.userId == request.auth.uid;
      allow read: if (isSignedIn() && request.auth.uid == resource.data.userId) || isAdmin();
      allow update: if (isSignedIn() && request.auth.uid == resource.data.userId) || isAdmin();
      allow delete: if isAdmin();
    }

    // Products: public read; only admins may write
    match /products/{productId} {
      allow read: if true;
      allow create, update, delete: if isAdmin();
    }

    // Test collection for connectivity checks
    match /test/{document} {
      allow read, write: if true;
    }

    // Deny all other access
    match /{document=**} {
      allow read, write: if false;
    }
  }
}

// How to publish:
// 1) Firebase Console → Firestore Database → Rules.
// 2) Paste these rules and click Publish.
